Represents an interface for a bounding box defined by top-right and bottom-left coordinates. This interface extends the IEntity interface and implements IEnumerable<T>.

public interface IBoundingBox<TCoordinates, TAngle> : IEntity, IEnumerable<TCoordinates>, IEnumerable where TCoordinates : ICoordinates<TAngle> where TAngle : IAngle
